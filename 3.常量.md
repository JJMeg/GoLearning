#### 关键字
- const
- 可作枚举
```Go
package main

import (
	"fmt"
	_ "fmt"
)

const (
	Level1 = 1
	Level2 = 2
	Level3 = 3
	level1 = iota
	level2
	level3
	error = 4
	level4
) //作枚举

func main() {
	const width = 5
	const length int = 2

	fmt.Println(width * length)

	const status = Level1
	fmt.Println(status)

	fmt.Println(level1, level2, level3, level4, error)
}
```
