#### 一个用例

```Go
package main

import (
	"fmt"
	_ "fmt"
)

func main() {
	var i int = 1;
	for true {
		i ++
		if (i > 10) {
			fmt.Println("loop1: ", i)
			break
		}
	}

	for j := 0; j < 10; j++ {
		fmt.Println("loop2: ", j)
	}

	for i < 20 {
		i++
		fmt.Println("loop3: ", i)
	}

	number := [6]int{1, 2, 3, 4, 5, 6}
	for k, x := range number {
		fmt.Println("loop4: index %d: %d", k, x)
	}

}
```

#### 循环的range写法
```Go
arr := []int {1,2,3}
for _, v := range arr {//只会循环切片中的三个元素，不会无限循环下去
  arr = append(arr, v) //追加的元素不会导致循环次数的增加
}
fmt.Println(arr)
```
